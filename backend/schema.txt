create table public.forms (
  id uuid not null default gen_random_uuid(),
  title character varying(255) not null,
  description text null,
  status character varying(50) not null default 'active',
  created_by uuid null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  constraint forms_pkey primary key (id),
  constraint forms_status_check check (
    (status)::text = any (array['active', 'inactive', 'archived']::text[])
  )
) tablespace pg_default;

create trigger forms_updated_at before
update on forms for each row
execute function update_updated_at_column();


create table public.form_fields (
  id uuid not null default gen_random_uuid(),
  form_id uuid not null,
  label character varying(255) not null,
  field_type character varying(50) not null default 'text',
  placeholder character varying(255) null,
  required boolean not null default false,
  options jsonb null default '[]'::jsonb,
  field_order integer not null default 0,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  constraint form_fields_pkey primary key (id),
  constraint form_fields_form_id_fkey foreign key (form_id) references forms (id) on delete cascade,
  constraint form_fields_type_check check (
    (field_type)::text = any (
      array['text', 'email', 'phone', 'number', 'textarea', 'select', 'checkbox', 'radio', 'file', 'date', 'url']::text[]
    )
  )
) tablespace pg_default;

create index if not exists idx_form_fields_form on public.form_fields using btree (form_id) tablespace pg_default;

create trigger form_fields_updated_at before
update on form_fields for each row
execute function update_updated_at_column();


create table public.form_responses (
  id uuid not null default gen_random_uuid(),
  form_id uuid not null,
  job_id uuid null,
  respondent_name character varying(255) null,
  respondent_email character varying(255) null,
  submitted_at timestamp with time zone not null default now(),
  constraint form_responses_pkey primary key (id),
  constraint form_responses_form_id_fkey foreign key (form_id) references forms (id) on delete cascade,
  constraint form_responses_job_id_fkey foreign key (job_id) references job_postings (id) on delete set null
) tablespace pg_default;

create index if not exists idx_form_responses_form on public.form_responses using btree (form_id) tablespace pg_default;
create index if not exists idx_form_responses_job on public.form_responses using btree (job_id) tablespace pg_default;


create table public.form_response_data (
  id uuid not null default gen_random_uuid(),
  response_id uuid not null,
  field_id uuid not null,
  value text null,
  constraint form_response_data_pkey primary key (id),

create table public.employees (

  email character varying(255) not null,
  phone character varying(50) null,
  department character varying(100) null,
  position character varying(100) null,
  manager character varying(255) null,
  join_date date null,
  status character varying(50) not null default 'active'::character varying,
  address text null,
  emergency_contact_name character varying(255) null,
  emergency_contact_relationship character varying(100) null,
  emergency_contact_phone character varying(50) null,
  bank_name character varying(255) null,

  bank_routing_number character varying(100) null,

  created_at timestamp with time zone not null default now(),

  location text null,
  alternate_phone text null,
  blood_group text null,
  exit_workflow jsonb null,
  constraint employees_pkey primary key (id),
  constraint employees_email_key unique (email),
  constraint employees_employee_code_key unique (employee_code),
  constraint employees_user_id_fkey foreign KEY (user_id) references auth.users (id) on delete CASCADE,
  constraint employees_status_check check (
    (
      (status)::text = any (
        (
          array[
            'active'::character varying,
            'inactive'::character varying,
            'on_leave'::character varying,
            'terminated'::character varying,
            'probation'::character varying,
            'onboarding'::character varying,
            'exit'::character varying
          ]
        )::text[]
      )
    )
  )
) TABLESPACE pg_default;
